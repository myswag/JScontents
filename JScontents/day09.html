7、编写一个函数  兼容 byClassName方法

     function getEleByClassName(cname){
         if( document.getElementsByClassName ){ //是高版本浏览器
             return document.getElementsByClassName(cname);
         }

         //如果是 低版本浏览器
         // 找到页面中所有的标签对象， 遍历这些标签，如果某个标签的className  ==  cname  说明找到了要操作的元素
         var all = document.getElementsByTagName("*");
         var arr = [];
         for( var i = 0 ; i < all.length ; i++ ){
             if( cname == all[i].className ){//找到了要操作的元素  i位置处的元素
                 arr.push( all[i] );
             }
         }
         return arr;
    }
